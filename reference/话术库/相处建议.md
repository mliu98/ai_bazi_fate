CREATE TABLE advice_3d (

  priority        INT DEFAULT 1,              -- 1=三维精确匹配（最高优先）
  char_count      INT,                        -- 正文字数（便于质检）
  created_at      TIMESTAMP DEFAULT NOW()
);

-- 建议索引
CREATE INDEX idx_wx ON advice_3d USING GIN(wx_tags);
CREATE INDEX idx_q  ON advice_3d USING GIN(q_tags);
CREATE INDEX idx_tier ON advice_3d USING GIN(tier_tags);
